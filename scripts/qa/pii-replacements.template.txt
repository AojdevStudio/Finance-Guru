# Applied to ALL blobs and commit messages via --replace-text/--replace-message
#
# INSTRUCTIONS: Copy this file to pii-replacements.txt (gitignored) and
# replace all <PLACEHOLDER> values with your actual PII before running
# git-filter-repo. DO NOT commit the filled-in version.
#
# ORDERING: More specific (longer) patterns are listed before less specific
# patterns within each category to prevent partial match conflicts.
# git-filter-repo applies replacements sequentially.
# =====================================================================

# Replace with actual API token (64 hex chars)
literal:<YOUR_API_TOKEN>==>REDACTED_API_TOKEN

# === HIGH: Personal Names (longer patterns first, case-sensitive) ===
# Capitalized forms (from documents, commit messages)
literal:<FULL_NAME>==>REDACTED_NAME
literal:<LAST_NAME>==>REDACTED_NAME
literal:<FIRST_NAME>==>REDACTED_NAME
# Combined username MUST come before separate lowercase parts
literal:<USERNAME_COMBINED>==>redacted_user
# Lowercase forms (from function names, variable names, filenames)
literal:<LAST_NAME_LOWER>==>redacted_name
literal:<FIRST_NAME_LOWER>==>redacted_name

# === HIGH: Email Addresses (full email before domain parts) ===
literal:<EMAIL_1>==>redacted@example.com
literal:<EMAIL_2>==>redacted@example.com

# === MEDIUM: Domain Names (with TLD before without) ===
literal:<DOMAIN_1>.com==>example.com
