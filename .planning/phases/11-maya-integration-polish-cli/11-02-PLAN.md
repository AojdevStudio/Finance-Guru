    const TOPICS = [
      {
        id: 'dividend-strategy',
        title: 'Dividend Strategy',
        description: 'Investment-first mindset, margin flywheel, DRIP mechanics, and income ETF selection',
        icon: '\uD83D\uDCC8', // chart increasing
        file: '../dist/dividend-strategy-explorer.html',
        concepts: 21  // approximate from Phase 9 data

        description: 'Delta, gamma, theta, vega and their impact on options pricing and strategy',
        icon: '\u0394', // Greek delta
        file: '../dist/options-greeks-explorer.html',
        concepts: 20
      },
      {
        id: 'risk-management',
        title: 'Risk Management',
        description: 'VaR, Sharpe ratio, position sizing, diversification, and tail risk controls',
        icon: '\uD83D\uDEE1\uFE0F', // shield
        file: '../dist/risk-management-explorer.html',
        concepts: 20
      }

    const STORAGE_PREFIX = 'fin-guru-explorer';
    const STORAGE_VERSION = 'v1';

    function getCompletionStats(topicId) {
      // Read localStorage from the explorer's persistence (Phase 10 pattern)
      try {
        const key = `${STORAGE_PREFIX}.${STORAGE_VERSION}.${topicId}`;
        const raw = localStorage.getItem(key);
        if (!raw) return null;

        const state = JSON.parse(raw);
        if (!state || !state.knowledge) return null;

        const knowledge = state.knowledge; // Map of nodeId -> knowledge level
        const total = Object.keys(knowledge).length;
        if (total === 0) return null;

        // Count non-unknown states (familiar + confident + mastered = assessed)
        const assessed = Object.values(knowledge).filter(
          (v) => v !== 'unknown'
        ).length;

        // Count mastered specifically
        const mastered = Object.values(knowledge).filter(
          (v) => v === 'mastered'
        ).length;

        return { total, assessed, mastered };
      } catch {
        return null;
      }
    }

    function renderCards() {
      const grid = document.getElementById('explorer-grid');
      if (!grid) return;

      grid.innerHTML = '';

      for (const topic of TOPICS) {
        const stats = getCompletionStats(topic.id);

        const card = document.createElement('a');
        card.className = 'topic-card';
        card.href = topic.file;

        // Badge
        let badge = '';
        if (stats) {
          const pct = Math.round((stats.assessed / stats.total) * 100);
          badge = `<span class="completion-badge">${pct}% assessed</span>`;
        } else {
          badge = `<span class="completion-badge new">New</span>`;
        }

        card.innerHTML = `
          <div class="card-icon">${topic.icon}</div>
          <div class="card-content">
            <h2>${topic.title}</h2>
            <p>${topic.description}</p>
            <div class="card-meta">
              <span class="concept-count">${topic.concepts} concepts</span>
              ${badge}
            </div>
          </div>
        `;

        grid.appendChild(card);
      }
    }

    document.addEventListener('DOMContentLoaded', renderCards);
  </script>
</body>
</html>

**Modify `src/cli/fin_guru.py`:**

The current file is a simple TUI launcher. Refactor to support subcommands: the default (no args) launches the TUI, and `explore` opens explorers.

```python
#!/usr/bin/env python3
"""
Finance Guru CLI Entry Point

WHAT: CLI entry point for Finance Guru dashboard and tools
WHY: Provides single command interface for all Finance Guru features
ARCHITECTURE: Subcommand router -- default launches TUI, 'explore' opens knowledge explorers

Usage:
    uv run python src/cli/fin_guru.py              # Launch TUI dashboard
    uv run python src/cli/fin_guru.py explore       # Open topic selector landing page
    uv run python src/cli/fin_guru.py explore <topic>  # Open specific explorer
    uv run python src/cli/fin_guru.py explore --list    # List available topics

Author: Finance Guru Development Team
Created: 2025-11-17
Updated: 2026-02-03
"""

import argparse
import sys
import webbrowser
from pathlib import Path

# Add project root to path
project_root = Path(__file__).parent.parent.parent
sys.path.insert(0, str(project_root))

# Registry of available explorer topics
# Maps slug -> display name for discovery
EXPLORER_TOPICS: dict[str, str] = {
    "dividend-strategy": "Dividend Investment Strategy",
    "options-greeks": "Options Greeks",
    "risk-management": "Risk Management",
}


def get_explorer_path(topic_slug: str) -> Path:
    """
    Get the filesystem path to an explorer HTML file.

    Args:
        topic_slug: Topic identifier (e.g., 'dividend-strategy')

    Returns:
        Path to the explorer HTML file

    Raises:
        FileNotFoundError: If the explorer file doesn't exist
    """
    explorer_path = project_root / "src" / "explorer" / "dist" / f"{topic_slug}-explorer.html"
    if not explorer_path.exists():
        raise FileNotFoundError(
            f"Explorer not found: {explorer_path}\n"
            f"Run the build pipeline first: bun run src/explorer/build.ts src/explorer/topics/{topic_slug}.json"
        )
    return explorer_path


def get_landing_path() -> Path:
    """
    Get the filesystem path to the topic selector landing page.

    Returns:
        Path to index.html

    Raises:
        FileNotFoundError: If the landing page doesn't exist
    """
    landing_path = project_root / "src" / "explorer" / "landing" / "index.html"
    if not landing_path.exists():
        raise FileNotFoundError(
            f"Landing page not found: {landing_path}\n"
            f"Ensure src/explorer/landing/index.html exists."
        )
    return landing_path


def cmd_explore(args: argparse.Namespace) -> None:
    """
    Handle the 'explore' subcommand.

    Opens a knowledge explorer in the default browser.
    With no topic: opens the topic selector landing page.
    With --list: prints available topics.
    With a topic slug: opens that specific explorer.
    """
    if args.list_topics:
        print("Available Knowledge Explorers:")
        print()
        for slug, name in EXPLORER_TOPICS.items():
            path = project_root / "src" / "explorer" / "dist" / f"{slug}-explorer.html"
            status = "built" if path.exists() else "not built"
            print(f"  {slug:<25} {name} [{status}]")
        print()
        print("Usage: uv run python src/cli/fin_guru.py explore <topic-slug>")
        return

    if args.topic is None:
        # No topic specified -- open landing page
        try:
            landing = get_landing_path()
            url = landing.as_uri()
            print(f"Opening topic selector: {landing}")
            webbrowser.open(url)
        except FileNotFoundError as e:
            print(f"Error: {e}", file=sys.stderr)
            sys.exit(1)
        return

    topic = args.topic

    if topic not in EXPLORER_TOPICS:
        available = ", ".join(EXPLORER_TOPICS.keys())
        print(
            f"Error: Unknown topic '{topic}'\n"
            f"Available topics: {available}\n"
            f"Run with --list for details.",
            file=sys.stderr,
        )
        sys.exit(1)

    try:
        explorer_path = get_explorer_path(topic)
        url = explorer_path.as_uri()
        print(f"Opening {EXPLORER_TOPICS[topic]} explorer: {explorer_path}")
        webbrowser.open(url)
    except FileNotFoundError as e:
        print(f"Error: {e}", file=sys.stderr)
        sys.exit(1)


def cmd_dashboard(args: argparse.Namespace) -> None:
    """Handle the default command -- launch TUI dashboard."""
    from src.ui.app import FinanceGuruApp

    app = FinanceGuruApp()
    app.run()


def parse_args() -> argparse.Namespace:
    """Parse command-line arguments with subcommands."""
    parser = argparse.ArgumentParser(
        description="Finance Guru CLI - Personal Financial Command Center",
        formatter_class=argparse.RawDescriptionHelpFormatter,
        epilog=__doc__,
    )

    subparsers = parser.add_subparsers(dest="command")

    # explore subcommand
    explore_parser = subparsers.add_parser(
        "explore",
        help="Open knowledge explorers in your browser",
        description="Launch interactive knowledge explorers for financial self-assessment",
    )
    explore_parser.add_argument(
        "topic",
        nargs="?",
        default=None,
        help="Topic slug (e.g., dividend-strategy). Omit to open topic selector.",
    )
    explore_parser.add_argument(
        "--list",
        dest="list_topics",
        action="store_true",
        help="List all available explorer topics",
    )

    return parser.parse_args()


def main() -> None:
    """Main entry point."""
    args = parse_args()

    if args.command == "explore":
        cmd_explore(args)
    else:
        # Default: launch TUI dashboard (preserves backward compatibility)
        cmd_dashboard(args)


if __name__ == "__main__":
    main()
```

**Key design decisions:**
- Subcommand architecture: `explore` is a subcommand, default (no args) still launches TUI
- `EXPLORER_TOPICS` registry maps slugs to display names -- single source of truth
- Uses `Path.as_uri()` for proper `file://` URL generation (cross-platform)
- `webbrowser.open()` uses the system default browser
- `--list` flag shows available topics with build status
- Invalid topic shows available options in error message
- Lazy import of `FinanceGuruApp` (only when TUI is needed) avoids import overhead for `explore`

**IMPORTANT: Backward compatibility**
- Running `uv run python src/cli/fin_guru.py` with no arguments MUST still launch the TUI dashboard
- The `explore` subcommand is additive -- it does not change existing behavior
